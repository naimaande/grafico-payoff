    <meta name="viewport" content="width=device-width, initial-scale=1.0">

    <title>Payoff CALL LONG</title>

    <script src="https://cdn.jsdelivr.net/npm/chart.js"></script>

</head>

<body>

    <h2>Grafico Payoff CALL LONG</h2>

    <canvas id="payoffChart"></canvas>


    <script>

        // Parametri dell'opzione

        const strikePrice = 100;  // Prezzo di esercizio (K)

        const premium = 10;       // Premio pagato (P)


        // Generazione dei dati per il grafico

        let prices = [];

        let payoffs = [];


        for (let S_T = 50; S_T <= 150; S_T += 2) {

            let payoff = Math.max(S_T - strikePrice, 0) - premium;

            prices.push(S_T);

            payoffs.push(payoff);

        }


        // Creazione del grafico con Chart.js

        const ctx = document.getElementById('payoffChart').getContext('2d');

        new Chart(ctx, {

            type: 'line',  // Tipo di grafico (linea)

            data: {

                labels: prices,  // Etichette (prezzi del sottostante)

                datasets: [{

                    label: 'Payoff CALL LONG',

                    data: payoffs,  // Dati di payoff

                    borderColor: 'blue',

                    borderWidth: 2,

                    fill: false

                }]

            },

            options: {

                responsive: true,

                scales: {

                    x: { title: { display: true, text: 'Prezzo del sottostante (S_T)' } },

                    y: { title: { display: true, text: 'Payoff' } }

                }

            }

        });

    </script>

</body>
